    /*
    *获取url参数并且返回值
    */
    function getQueryString(name, url){
        let search = url || window.location.search
        let reg = new RegExp("(^|&)" + name + "=([^&]*)(&|$)", "i");
        let r = search.substr(1).match(reg);
        if (r !== null) return decodeURIComponent(r[2]);
        return null;
    }
